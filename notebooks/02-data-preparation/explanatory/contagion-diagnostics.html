7. Coverage map

Which ADM1 units are in the panel, and how many strike onsets did each have?

Code 
strike_counts <- panel |>
  group_by(gid_1) |>
  summarize(n_onset = sum(strike_onset), .groups = "drop")

map_data <- adm1_sf |>
  left_join(strike_counts, by = c("GID_1" = "gid_1")) |>
  mutate(
    status = case_when(
      is.na(n_onset) ~ "Not in panel",
      n_onset == 0   ~ "In panel, no strikes",
      TRUE           ~ "In panel, ≥1 strike"
    ),
    n_onset_log = log1p(n_onset)
  )

ggplot(map_data) +
  geom_sf(aes(fill = n_onset_log), linewidth = 0.05, color = "white") +
  geom_sf(
    data = map_data |> filter(is.na(n_onset)),
    fill = "#d9d9d9", linewidth = 0.05, color = "white"
  ) +
  scale_fill_gradient(
    low = "#c6dbef", high = "#08306b",
    na.value = "#d9d9d9",
    name = "Strike\nonsets\n(log scale)",
    labels = \(x) round(expm1(x))
  ) +
  labs(
    title = "Strike onset frequency by ADM1 region",
    subtitle = "Grey = not in contagion panel"
  ) +
  theme_minimal() +
  theme(
    legend.position  = "right",
    axis.text        = element_blank(),
    panel.grid.major = element_line(color = "grey90", linewidth = 0.2)
  )

# Highlight isolates (no within-country neighbors)
adj <- readRDS(here("data/spatial/gadm_adjacency.rds"))
isolates <- adj$summary |> filter(n_neighbors_within == 0)

map_isolates <- adm1_sf |>
  filter(GID_1 %in% unique(panel$gid_1)) |>
  mutate(is_isolate = GID_1 %in% isolates$gid_1)

ggplot(map_isolates) +
  geom_sf(aes(fill = is_isolate), linewidth = 0.05, color = "white") +
  scale_fill_manual(
    values = c("FALSE" = "#c6dbef", "TRUE" = "firebrick"),
    labels = c("FALSE" = "Has neighbors", "TRUE" = "Isolate (no within-country neighbor)"),
    name = NULL
  ) +
  labs(
    title = "ADM1 isolates: units with no within-country neighbors",
    subtitle = "Isolates receive NA exposure and are excluded from main specification"
  ) +
  theme_minimal() +
  theme(
    legend.position  = "bottom",
    axis.text        = element_blank(),
    panel.grid.major = element_line(color = "grey90", linewidth = 0.2)
  )
